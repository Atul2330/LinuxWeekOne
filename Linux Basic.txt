# Creating a textfile.txt to manipute its permission
touch textfile.txt
ls -l
nano textfile.txt
./textfile.txt (won't work as we dont have permission to execute )
chmod u+wrx textfile.txt || chmod u+x textfile.txt
./textfile.txt (works)

#Changing ownership of this textfile.txt to atuls
sudo chown atuls textfile.txt (changes ownership to atuls)
sudo chown atul textfile.txt (changes ownership back to atul)
